package problem1;

import java.util.Objects;

/**
/**
 * A class representing non empty node.
 */
public class ElementNode implements ILinkedList{
  private Integer priority;
  private String priorityValue;
  private ILinkedList rest;

  /**
   * constructor for the non empty linked list node.
   * @param priority Priority
   * @param priorityValue The value associated with priority.
   * @param rest The next ILinkedList.
   */
  public ElementNode(Integer priority, String priorityValue, ILinkedList rest) {
    this.priority = priority;
    this.priorityValue = priorityValue;
    this.rest = rest;
  }

  /**
   * Counts number of non empty nodes.
   * @return number of non empty nodes.
   */
  @Override
  public Integer count() {
    return 1 + this.rest.count() ;
  }

  /**
   * Gets the rest linked list.
   * @return the rest of linked list.
   */
  @Override
  public ILinkedList getRest() {
    return this.rest;
  }

  /**
   * Set the rest linked list.
   * @param rest The rest linked list.
   */
  @Override
  public void setRest(ILinkedList rest) {
    this.rest = rest;
  }

  /**
   * Gets the priority of the element.
   * @return the priority of the node.
   */
  @Override
  public Integer getPriority() {
    return this.priority;
  }
//

  /**
   * Gets the value associated with the priority.
   * @return the value of the priority.
   */
  @Override
  public String getPriorityValue() {
    return this.priorityValue;
  }

  /**
   * Override equals method.
   * @param o Object to be compared.
   * @return true if attributes are equals.
   */
  @Override
  public boolean equals(Object o) {
    if (this == o) {
      return true;
    }
    if (!(o instanceof ElementNode)) {
      return false;
    }
    ElementNode that = (ElementNode) o;
    return Objects.equals(priority, that.priority) && Objects
        .equals(priorityValue, that.priorityValue) && Objects.equals(rest, that.rest);
  }

  /**
   * Override hashcode method.
   * @return true if hashcode are equals.
   */
  @Override
  public int hashCode() {
    return Objects.hash(priority, priorityValue, rest);
  }

  /**
   * Override toString method.
   * @return true if Strings are equal.
   */
  @Override
  public String toString() {
    return "ElementNode{" +
        "priority=" + priority +
        ", priorityValue='" + priorityValue + '\'' +
        ", rest=" + rest +
        '}';
  }

}
